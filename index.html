    ② Apps Script
    function doGet(e) { const id=e.parameter.id, pw=e.parameter.pw; const s=SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("Sheet1"); const d=s.getDataRange().getValues(); const h=d[0]; for(let i=1;i<d.length;i++){ if(String(d[i][0])===id && String(d[i][1])===pw){ const sc=[]; for(let j=4;j<h.length;j+=3) if(h[j]) sc.push({label:h[j],score:d[i][j+1],total:d[i][j+2]}); return out({ok:true,name:d[i][2],subject:d[i][3],scores:sc}); } } return out({ok:false}); } function out(o){ return ContentService.createTextOutput(JSON.stringify(o)) .setMimeType(ContentService.MimeType.JSON); }
